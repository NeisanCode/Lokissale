$data = [
    "pays" => $pays,
    "ville" => $ville,
    "adresse" => $adresse,
    "cp" => $cp,
    "titre" => $titre,
    "description" => $description,
    "capacite" => $capacite,
    "categorie" => $categorie,
    "photo" => $photo,
];

// 1. On prépare les morceaux de la requête : "colonne = ?"
$champs = [];
foreach ($data as $colonne => $valeur) {
    $champs[] = "$colonne = ?";
}

// 2. On assemble la requête finale
$sql = "UPDATE salles SET " . implode(", ", $champs) . " WHERE id_salle = ?";

// 3. On prépare les valeurs (les valeurs du dictionnaire + l'ID pour le WHERE)
$valeurs = array_values($data);
$valeurs[] = $id_salle; // L'ID de la salle que vous voulez modifier

// 4. Exécution sécurisée
$stmt = $pdo->prepare($sql);
$stmt->execute($valeurs);
